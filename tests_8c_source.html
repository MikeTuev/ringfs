145 
146  /* first format a filesystem */

151  ringfs_format(&fs);
152 
153  /* now we're brave and we write some data */
154  for (int i=0; i<3; i++) {
155  printf("## ringfs_append()\n");
156  ringfs_append(&fs, (int[]) { 0x11*(i+1) });
157 
158  /* make sure the write head has advanced */
159  ck_assert_int_eq(fs.write.slot, i+1);
160  }
161 
162  /* now we fetch at it. */
163  int obj;
164  for (int i=0; i<3; i++) {
165  printf("## ringfs_fetch()\n");
166  ck_assert(ringfs_fetch(&fs, &obj) == 0);
167  ck_assert_int_eq(obj, 0x11*(i+1));
168 
169  /* make sure the cursor head has advanced */
170  ck_assert_int_eq(fs.cursor.slot, i+1);
171  }
172  /* there should be no data left */

174
